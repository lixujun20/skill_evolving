```py
from typing import Any, Dict
from pydantic import Field
from app.old_tool.base import MadeTool, ToolResult
from app.llm import LLM
from app.schema import Message

class TextProcessingTool(MadeTool):
    name: str = "TextProcessingTool"
    description: str = (
        "A tool for processing text. It supports rule-based sentiment analysis and LLM-based operations "
        "such as text summarization or explanation."
    )
    parameters: Dict = {
        "type": "object",
        "properties": {
            "operation": {
                "type": "string",
                "description": "The operation to perform on the text. Must be one of: 'sentiment_analysis', 'summarization', 'explanation'.",
                "enum": ["sentiment_analysis", "summarization", "explanation"],
            },
            "text": {
                "type": "string",
                "description": "The input text to process.",
            },
            "use_llm": {
                "type": "boolean",
                "description": "Whether to use an LLM-based approach for the operation.",
            },
        },
        "required": ["operation", "text"],
    }
    output_schema: Dict = {
        "type": "object",
        "properties": {
            "result": {
                "type": "string",
                "description": "The processed text."
            },
        },
    }

    # Field for integrating the LLM
    llm: LLM = Field(default_factory=LLM)

    async def execute(self, **kwargs) -> ToolResult:
        operation = kwargs.get("operation")
        text = kwargs.get("text")
        use_llm = kwargs.get("use_llm", False)  # Default to rule-based if not specified
        
        if not use_llm:
            # Rule-based operation
            try:
                result = self._rule_based_processing(operation, text)
                return ToolResult(output=json.dumps({"result": result}))
            except Exception as e:
                return ToolResult(error=str(e))
        else:
            # LLM-based processing
            try:
                llm_request = self._build_llm_request(operation, text)
                llm_result = await self.llm.ask(user=[Message.user_message(llm_request)])
                return ToolResult(output=json.dumps({"result": llm_result}))
            except Exception as e:
                return ToolResult(error=str(e))

    def _rule_based_processing(self, operation: str, text: str) -> Any:
        """
        Perform textual operations using predefined rules.
        """
        if operation == "sentiment_analysis":
            return self._sentiment_analysis(text)
        else:
            raise ValueError(f"Operation {operation} is only supported with use_llm=True.")

    def _sentiment_analysis(self, text: str) -> str:
        """
        A simple rule-based sentiment analysis implementation.
        """
        positive_keywords = ["good", "great", "happy", "excellent"]
        negative_keywords = ["bad", "sad", "terrible", "poor"]

        text_lower = text.lower()
        positive_score = sum(text_lower.count(word) for word in positive_keywords)
        negative_score = sum(text_lower.count(word) for word in negative_keywords)

        if positive_score > negative_score:
            return "Positive sentiment detected in the text."
        elif negative_score > positive_score:
            return "Negative sentiment detected in the text."
        else:
            return "Neutral sentiment detected in the text."

    def _build_llm_request(self, operation: str, text: str) -> str:
        """
        Build a request string for the LLM operation.
        """
        if operation == "summarization":
            return f"Summarize the following text: {text}"
        elif operation == "explanation":
            return f"Explain the meaning or context of the following text: {text}"
        else:
            raise ValueError(f"Invalid operation for LLM-based processing: {operation}")
```
